#!/bin/bash

# Color codes for ANSI color escape squences
RESET_COLOR=0
RED=31
GREEN=32
NO_COLOR=33
BLUE=34
MAGENTA=35
CYAN=36

ANNOUNCE_COLOR="$BLUE"
ERROR_COLOR="$RED"
INFO_COLOR="$GREEN"
CODEBLOCK_COLOR="$BLUE"

function text_color() {
  #{{- if eq .Values.test.colorize true }}
  color_code="$1"
  echo -e '\033[0;'"$color_code"'m'
  #{{- else }}
  echo
  #{{- end }}
}

function banner() {
  text_color "$1"
  shift
  echo =====================================================================
  echo -e "${@}"
  echo =====================================================================
  text_color "$RESET_COLOR"
}

function indented_banner() {
  text_color "$1"
  shift
  echo "      --------------------------------------------------------------"
  echo -e "      ${@}"
  echo "      --------------------------------------------------------------"
  text_color "$RESET_COLOR"
}

function announce() {
  banner "$BLUE" "$@"
}

function display_error() {
  banner "$RED" "$@"
}

function display_info() {
  text_color "$INFO_COLOR"
  echo -e "$@"
  text_color "$RESET_COLOR"
}

function codeblock() {
  indented_banner "$CODEBLOCK_COLOR" "$@"
}
